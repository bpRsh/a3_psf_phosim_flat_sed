    for i in range(21):
        
        infile = indir + '/psf{}.fits'.format(i)
        data   = fits.getdata(infile)
        shape  = data.shape
    
        rows = data.shape[0]
        total = 0.0
        for i in range(rows):
            total += sum(data[i])
        print('{} {} {}'.format(infile,' sum_of_all_pixels = ',total))
